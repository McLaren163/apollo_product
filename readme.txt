Программа для генерации чертежей калиток

D:\apollo_product\apollo_doors.bat


запуск с параметрами:

	-b "Заказчик"
	-o "Номер заказа"
	-e "Инженер"
	-cfr "Цвет рамы"
	-cfi "Цвет заполнения"
	-W "Ширина калитки"
	-H "Высота калитки"
	-C "Просвет"
	-br "Перемычка" ("да", "нет", "фрамуга")
	-brh "Высота фрамуги"
	-open "Открытие" ("на улицу", "во двор")
	-side "Сторона петель" ("слева", "справа")


Пример:

D:\apollo_product\apollo_doors.bat -b "Иванов Петр" -o 117 -e "Баганов Антон" -cfr 8017 -cfi 9016 -W 1000 -H 2500 -C 50 -br "фрамуга" -brh 300 -open "на улицу" -side слева
